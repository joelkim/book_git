style_git_region = r"""
    draw=black!70,
    fill=lightgray!10,
    minimum width=4.5cm,
    minimum height=4.5cm,
    rectangle,
    line width=2pt
"""
style_git_file = r"""
    draw=black!70,
    fill=lightgray!20,
    minimum width=3cm,
    minimum height=0.7cm,
    rectangle,
    line width=1pt
"""
style_git_operation = r"""
    draw=black!70,
    -Triangle,
    line width=1.2pt,
    bend left=80,
    looseness=1.4
"""
style_git_pointer = r"""
    draw=black!70,
    -Triangle,
    line width=1.2pt
"""
